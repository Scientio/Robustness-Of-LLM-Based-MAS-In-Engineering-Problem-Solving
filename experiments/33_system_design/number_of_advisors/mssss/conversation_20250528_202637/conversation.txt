```python
import matplotlib.pyplot as plt

# Data
reynolds_numbers = [1000, 1500, 2000, 2500, 3000, 3500, 4000, 4500, 5000]
pressure_loss_steel = [0.305, 0.357, 0.410, 1.600, 2.140, 2.790, 3.350, 4.220, 4.730]

pressure_loss_copper = [0.305, 0.392, 0.398, 0.493, 0.619, 0.752, 0.944, 1.177, 1.177]

# Create the plot
plt.figure(figsize=(10, 6))
plt.plot(reynolds_numbers, pressure_loss_steel, marker='o', label='Commercial Steel', color='blue')
plt.plot(reynolds_numbers, pressure_loss_pvc, marker='o', label='PVC', color='green')
plt.plot(reynolds_numbers, pressure_loss_copper, marker='o', label='Copper', color='orange')

# Adding titles and labels
plt.title('Pressure Loss Across Different Materials')
plt.xlabel('Reynolds Number (Re)')
plt.ylabel('Pressure Loss (ΔP) [Pa]')
plt.grid(True)
plt.legend()
plt.xticks(reynolds_numbers)  # Set x-ticks to Reynolds numbers for clarity

# Show the plot
plt.tight_layout()
plt.show()
```
